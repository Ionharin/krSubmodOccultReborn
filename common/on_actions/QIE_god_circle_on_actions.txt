on_actions = {
	on_daily = {
		effect = {
			if = {
				limit = {
					tag = QIE
					date > 1936.11.1
					date < 1936.11.30
					NOT = { has_country_flag = QIE_ws_leader_30_triggered }
				}
				set_country_flag = QIE_ws_leader_30_triggered
				country_event = { id = QIE_ws_leader.30 days = 0 }
			}
			if = {
				limit = {
					tag = QIE
					NOT = { has_country_flag = QIE_ws_normal_event_10_triggered }
					NOT = { country_exists = HUN }
					HUN = {
						any_core_state = { is_controlled_by = QIE }
						all_core_state = { is_controlled_by = QIE }
					}
				}
				set_country_flag = QIE_ws_normal_event_10_triggered
				country_event = { id = QIE_ws_normal_events.10 days = 0 }
			}
			if = {
				limit = {
					tag = QIE
					has_completed_focus = QIE_seances_restart_in_beijing
					has_variable = god_circle_focus_var
					check_variable = { god_circle_tier < 3 }
					NOT = {
						has_country_flag = QIE_recently_changed_god_circle_focus
					}
				}
				add_to_variable = {
					god_circle_tier = 1
				}
				set_country_flag = {
					flag = QIE_recently_changed_god_circle_focus
					days = 70
					value = 1
				}
				QIE_update_god_circle_tier = yes
			}
		}
	}
}
