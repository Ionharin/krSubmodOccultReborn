# Some scripted triggers to analyze the Chinese Factions in QIE WUSHAN PNR

china_character_is_traditionalist_china = { # wushan
	event_target:system = { china_alignment_traditionalist_china = { check_variable = { PREV.PREV.china_alignment = THIS } } }
}

china_alignment_traditionalist_china = { #wushan
    custom_override_tooltip = {
        tooltip = china_alignment_traditionalist_china
        var:country_leader = { china_character_is_traditionalist_china = yes }
    }
}

china_is_aligned_with_traditionalist_china = { #wushan
    custom_override_tooltip = {
        tooltip = china_is_aligned_with_traditionalist_china
        OR = {
            china_alignment_traditionalist_china = yes
            has_cosmetic_tag = QIE_wushan_cosmetic_tag
            has_cosmetic_tag = QIE_united_china_ws
            has_cosmetic_tag = QIE_united_china_ws_final_phase
            has_country_flag = QIE_wushan_orthodoxy_set_flag
            has_country_flag = qie_wushan_has_control_china
        }
    }
}

china_compatible_alignment_to_PREV = {
	hidden_trigger = {
		china_is_unaligned = no
		PREV = { china_is_unaligned = no }
	}
	OR = {
		china_same_alignment_as_PREV = yes
		AND = {
			china_is_aligned_with_zhili = yes
			PREV = { china_is_aligned_with_zhili = yes }
		}
		AND = {
			china_is_aligned_with_fengtian = yes
			PREV = { china_is_aligned_with_fengtian = yes }
		}
		AND = {
			china_is_aligned_with_manchu = yes
			PREV = { china_is_aligned_with_manchu = yes }
		}
		AND = {
			china_is_aligned_with_federalists = yes
			PREV = { china_is_aligned_with_federalists = yes }
		}
		AND = {
			china_is_aligned_with_kmt = yes
			PREV = { china_is_aligned_with_kmt = yes }
		}
		AND ={ # wushan
			china_is_aligned_with_traditionalist_china = yes
			PREV = { china_is_aligned_with_traditionalist_china = yes }
		}
	}
}

china_compatible_alignment_to_FROM = {
	hidden_trigger = {
		china_is_unaligned = no
		FROM = { china_is_unaligned = no }
	}
	OR = {
		FROM = { check_variable = { country_leader:china_alignment = PREV.country_leader:china_alignment } }
		AND = {
			china_is_aligned_with_zhili = yes
			FROM = { china_is_aligned_with_zhili = yes }
		}
		AND = {
			china_is_aligned_with_fengtian = yes
			FROM = { china_is_aligned_with_fengtian = yes }
		}
		AND = {
			china_is_aligned_with_manchu = yes
			FROM = { china_is_aligned_with_manchu = yes }
		}
		AND = {
			china_is_aligned_with_federalists = yes
			FROM = { china_is_aligned_with_federalists = yes }
		}
		AND = {
			china_is_aligned_with_kmt = yes
			FROM = { china_is_aligned_with_kmt = yes }
		}
		AND = { # wushan
			china_is_aligned_with_traditionalist_china = yes

		}
	}
}
