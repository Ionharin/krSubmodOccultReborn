# Henan line: project -> develop -> final
country_event = {
	id = QIE_wushan_auto_industry.01
	hidden = yes
	is_triggered_only = yes
	trigger = { has_completed_focus = QIE_pauper_factory }
	immediate = {
		if = {
			limit = { NOT = { has_completed_focus = QIE_project_henan } }
			activate_shine_on_focus = QIE_project_henan
			set_country_flag = { flag = QIE_project_henan_in_progress_flag days = 35 value = 1 }
			country_event = { id = QIE_wushan_auto_industry.02 days = 35 }
		}
		else_if = {
			limit = { NOT = { has_completed_focus = QIE_develop_henan } }
			activate_shine_on_focus = QIE_develop_henan
			set_country_flag = { flag = QIE_develop_henan_in_progress_flag days = 105 value = 1 }
			country_event = { id = QIE_wushan_auto_industry.02 days = 105 }
		}
		else_if = {
			limit = { NOT = { has_completed_focus = QIE_final_industry_henan } }
			activate_shine_on_focus = QIE_final_industry_henan
			set_country_flag = { flag = QIE_final_industry_henan_in_progress_flag days = 133 value = 1 }
			country_event = { id = QIE_wushan_auto_industry.02 days = 133 }
		}
	}
}
