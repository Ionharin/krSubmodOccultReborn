add_namespace = QIE_wushan_wu_jiang_power_auto

country_event = {
	id = QIE_wushan_wu_jiang_power_auto.01
	hidden = yes
	is_triggered_only = yes
	trigger = {
		has_completed_focus = QIE_wushan_power
	}
	immediate = {
		if = {
			limit = {
				NOT = { has_completed_focus = QIE_the_rise_of_jiang }
			}
			activate_shine_on_focus = QIE_the_rise_of_jiang
			set_country_flag = {
				flag = QIE_the_rise_of_jiang_in_progress_flag
				days = 49
				value = 1
			}
			country_event = { id = QIE_wushan_wu_jiang_power_auto.02 days = 49 }
		}
		else_if = {
			limit = {
				NOT = { has_completed_focus = QIE_lobby_members }
			}
			activate_shine_on_focus = QIE_lobby_members
			set_country_flag = {
				flag = QIE_lobby_members_in_progress_flag
				days = 21
				value = 1
			}
			country_event = { id = QIE_wushan_wu_jiang_power_auto.02 days = 21 }
		}
	}
}
